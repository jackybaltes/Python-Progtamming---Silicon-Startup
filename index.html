def find_max_subsequence(lst):
  #print( f'fms {lst}' ) 
  max = None
  maxRange = (None,None)
  for start in range( len(lst ) ):
    for end in range( start + 1, len(lst) + 1 ):
      psum = sum( lst[start:end] )
      #print( f'sum {psum} max {max}, range {maxRange}')
      if max is None or psum > max:
        max = psum
        maxRange = (start, end)
  return maxRange, max

Prefix Sums

  • Algorithm currently has O(n**2) time complexity
  • Can be reduced to a linear complexity O(n) through the use of prefix sums
  • Implement a version using prefix sums

Prefix Sums

  • Initialize start to 0
  • Keep track of sum from start to current
  • Remember maximum of sum
  • If sum is less than 0, then reset sum to 0 and set start to next element

function find_max_subsequence( numbers : number[] ) : number[] {
    let msum = 0;
    let mmax = -Infinity;
    let start = 0;
    let mstart = -1;
    let mend = -1;

    for( let i = 0; i < numbers.length; i++ ) {
        let n = numbers[i]
        msum = msum + n;
        if ( msum > mmax ) {
            mmax = msum;
            mstart = start;
            mend = i;
        }
        if ( msum < 0 ) {
            msum = 0;
            start = i+1;
        }
    }
    return [ mmax, mstart, mend + 1];
}
